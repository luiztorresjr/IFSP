DROP TABLE func;
---------------------------------------------------------------------------------------------------------------------------------------------------
CREATE TABLE func(
	id int,
	nome varchar(20),
	chefe int,	
	data_adm date,
	FOREIGN KEY (chefe) REFERENCES func (id),
	PRIMARY KEY(id)
	);

SELECT f.nome AS nome_func
	,c.nome AS nome_chefe
FROM 	func AS c
	,func AS f
WHERE c.id = f.chefe;
---------------------------------------------------------------------------------------------------------------------------------------------------
SELECT f.nome AS nome_func
	,c.nome AS nome_chefe
FROM 	func AS c
INNER JOIN func AS f
	ON (f.chefe = c.id);
----------------------------------------------------------------------------------------------------------------------------------------------------
SELECT f.nome AS nome_func
	,c.nome AS nome_chefe
FROM 	func AS c
RIGHT JOIN func AS f
	ON (f.chefe = c.id);
